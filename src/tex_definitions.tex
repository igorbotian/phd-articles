\newcounter{SectionCounter}
\setcounter{SectionCounter}{1}

\newcounter{SubsectionCounter}
\setcounter{SubsectionCounter}{1}

% =============================================================================

\newcommand{\ClearPageStyle} {%
	\pagestyle{fancy}
	\fancyhf{}
	\renewcommand{\headrulewidth}{0pt}
}

\newcommand{\ApplyCommonPageStyle} {%
	\ClearPageStyle
	\cfoot{\thepage}
}

\newcommand{\Section} [1] {%
	\textbf{\arabic{SectionCounter}. #1}%
	\stepcounter{SectionCounter}%
	\setcounter{SubsectionCounter}{1}%
	%\label{sec:#2}%
}

\newcommand{\Subsection} [1] {%
	\textbf{\arabic{SectionCounter}.\arabic{SubsectionCounter}. #1}%
	\stepcounter{SubsectionCounter}%
	%\label{sec:#2}%
}

% =============================================================================

\newcommand{\sub} [1] {%
	\scriptsize{#1}%
}

\newcommand{\Rho}{%
	\mathcal{P}%
}

\newcommand{\link} [1] {%
	\textcolor{blue}{#1}%
}

\newcommand{\reference} [1] {%
	\link{\cite{#1}}%
}

\newcommand{\todo} [1] {%
	\textcolor{red}{#1}%
}
